<!-- Exact scoring formulas — extracted from src/lib/types.ts. Do not re-derive. -->

berekenComplexiteit(a: Automatisering): number
  stappenScore   = Math.min(stappen.length * 10, 40)
  systemenScore  = Math.min(systemen.length * 12, 36)
  afhankScore    = afhankelijkheden.trim() ? 15 : 0
  koppScore      = Math.min(koppelingen.length * 5, 15)
  total          = Math.min(stappenScore + systemenScore + afhankScore + koppScore, 100)

berekenImpact(a: Automatisering, alle: Automatisering[]): number
  fasenScore     = fasen.length * 12
  systemenScore  = systemen.length * 8
  depScore       = (count of automations in `alle` where koppelingen[].doelId === a.id) * 20
  statusBonus    = status === "Actief" ? 10 : 0
  total          = Math.min(fasenScore + systemenScore + depScore + statusBonus, 100)

  CRITICAL PITFALL: always pass a populated `alle` array to test depScore.
  If `alle` is empty or does not include the dependent automation, depScore = 0.

detectProblems(automations: Automatisering[]): GraphProblem[]
  Returns GraphProblem[] where each has: { id, automationId, type: ProblemType, severity, message, suggestion }

  ProblemType values: "orphan" | "missing-owner" | "missing-trigger" |
    "missing-systems" | "outdated" | "unverified" | "no-goal" | "broken-link"

  CRITICAL PITFALL (orphan): a single automation with empty koppelingen IS flagged as orphan.
    buildAdjacency marks any node with zero edges as orphan.
    To test "no orphan" behavior, create TWO automations with a koppelingen link between them.

  Koppeling type: { doelId: string; label?: string }
    Set koppeling.doelId to the id of the target automation.
